\section{Set Union $\cup$}

\begin{center}
$\cup \lbrack l_i,l_j\rbrack = l_i \cup l_j = \{l_i\} \cup \{l_j\} :=$

$\cup \lbrack l_i,l_j\rbrack \rightarrow \{l_i\} \hspace{2mm} i = j$
\\
$\cup \lbrack l_i,l_j\rbrack \rightarrow \{l_i,l_j\} \hspace{2mm} i \neq j$
\end{center}
